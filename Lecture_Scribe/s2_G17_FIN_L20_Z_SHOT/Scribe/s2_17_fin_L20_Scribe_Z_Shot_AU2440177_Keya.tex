\section{Random Vector Representation}

\[
X =
\begin{bmatrix}
X_1 \\
X_2 \\
\vdots \\
X_N
\end{bmatrix}
\]

\section{Mean Vector:}

\[
\mu_X = E[X]
\]

\[
\mu_X =
\begin{bmatrix}
E[X_1] \\
E[X_2] \\
\vdots \\
E[X_N]
\end{bmatrix}
\]

\section{Correlation Matrix:}

\[
R_{XX} = E[XX^T], \quad R_{XX}(i,j) = E[X_i X_j]
\]

\section{Covariance Matrix:}

\[
C_{XX} = E[(X - \mu_X)(X - \mu_X)^T]
\]

\section{Linear Transformation:}

\[
Y = AX + b
\]

\section{Mean Transformation:}


\[
\mu_Y = A\mu_X + b
\]


\section{Correlation Transformation:}

\[
R_{YY} = E[(AX + b)(AX + b)^T]
\]

\[
R_{YY} = A R_{XX} A^T + A\mu_X b^T + b\mu_X^T A^T + bb^T
\]

\section{Covariance Transformation:}

\[
Y - \mu_Y = A(X - \mu_X)
\]


\[
C_{YY} = A C_{XX} A^T
\]

\section{Summary:}

\[
\mu_Y = A\mu_X + b
\]

\[
C_{YY} = A C_{XX} A^T
\]

\[
R_{YY} = A R_{XX} A^T + A\mu_X b^T + b\mu_X^T A^T + bb^T
\]

\section{Gaussian Distribution:}

\subsection*{Univariate}

\[
f_X(x) = \frac{1}{\sqrt{2\pi\sigma^2}} 
\exp\left(-\frac{(x-\mu)^2}{2\sigma^2}\right)
\]

\subsection*{Joint Gaussian}

\[
f_{XY}(x,y) =
\frac{1}{2\pi\sigma_X \sigma_Y \sqrt{1-\rho^2}}
\exp\left(
-\frac{1}{2(1-\rho^2)}
\left[
\left(\frac{x-\mu_X}{\sigma_X}\right)^2
-2\rho\left(\frac{x-\mu_X}{\sigma_X}\right)\left(\frac{y-\mu_Y}{\sigma_Y}\right)
+\left(\frac{y-\mu_Y}{\sigma_Y}\right)^2
\right]
\right)
\]

\section{Multivariate Gaussian:}

\[
f_X(x) =
\frac{1}{\sqrt{(2\pi)^N \det(C_{XX})}}
\exp\left(
-\frac{1}{2}(x-\mu_X)^T C_{XX}^{-1}(x-\mu_X)
\right)
\]

\section{Uncorrelated Case:}

\[
Cov(X_i, X_j) = 0, \quad i \ne j
\]

\[
f_X(x) = \prod_{n=1}^{N}
\frac{1}{\sqrt{2\pi\sigma_n^2}}
\exp\left(-\frac{(x_n-\mu_n)^2}{2\sigma_n^2}\right)
\]
